centos-init-for-cm4.sh


conf_net() {
ipre=$1
hostno=$2

rm -f /etc/udev/rules.d/70-persistent-net.rules

cat > /etc/sysconfig/network << EOF
NETWORKING=yes
HOSTNAME=n$hostno.ex.com
GATEWAY=$ipre.1
EOF

cat > /etc/sysconfig/network-scripts/ifcfg-eth0 << EOF
DEVICE=eth0
ONBOOT=yes
BOOTPROTO=static
IPADDR=$ipre.10$hostno
NETMASK=255.255.255.0
EOF

cat > /etc/resolv.conf << EOF
nameserver $ipre.1
EOF

cat > /etc/hosts << EOF
127.0.0.1       localhost
$ipre.101 n1.ex.com n1
$ipre.102 n2.ex.com n2
$ipre.103 n3.ex.com n3
$ipre.104 n4.ex.com n4
$ipre.1 repo
EOF
}

conf_yum() {
rm /etc/yum.repos.d/*
cat > /etc/yum.repos.d/local.repo << "EOF"
[local-cm]
baseurl=http://repo/cdh/cm/4
gpgcheck=0

[local-cdh]
baseurl=http://repo/cdh/cdh/4
gpgcheck=0

[local-centos]
baseurl=http://repo/cdh/centos
gpgcheck=0
EOF
}

turnoff_se() {
chkconfig iptables off
cat > /etc/selinux/config << "EOF"
SELINUX=disabled
SELINUXTYPE=targeted
EOF
}

if [ -z "$1" ]; then
    echo 'a host no needed!'
else
    turnoff_se
    conf_yum
    conf_net '192.168.18' $1

    yum -y install perl openssh-clients postgresql-server bind-utils libxslt cyrus-sasl-gssapi parted vim wget
    mv /usr/bin/host /usr/bin/host.bak

    wget http://repo/cdh/cmi.bin
    chmod +x cmi.bin
    echo './cmi.bin --skip_repo_package=1' > inscm 
    chmod +x inscm
fi
