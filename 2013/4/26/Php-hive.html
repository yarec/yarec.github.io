php-hive

#!/usr/bin/env php

<?php

<<<EOF
work on hiveserver(1)

gen-thrift: 
http://blog.csdn.net/hzcyclone/article/details/7708327
svn co http://svn.apache.org/repos/asf/hive/tags/release-0.10.0
find ./ -name "*.thrift" | xargs -I{} cp {} thrift
cp thrift-src/fb303.thrift thrift
thrift -r --gen php hive_metastore.thrift
thrift -r --gen php hive_service.thrift

use in phpHiveAdmin/libs
https://github.com/xianglei/phpHiveAdmin.git

hive --service hiveserver -v -p 10001

EOF;

$GLOBALS['THRIFT_ROOT'] = 'libs/';
require_once $GLOBALS['THRIFT_ROOT'] . 'packages/hive_service/ThriftHive.php';
require_once $GLOBALS['THRIFT_ROOT'] . 'transport/TSocket.php';
require_once $GLOBALS['THRIFT_ROOT'] . 'protocol/TBinaryProtocol.php';

$transport = new TSocket('192.168.1.72', 10001);
$protocol = new TBinaryProtocol($transport);
$client = new ThriftHiveClient($protocol);
$transport->open();

#$client->execute('SELECT * from sample_07 limit 3');
$client->execute('show tables');
var_dump($client->fetchAll());

var_dump($client->getClusterStatus());

$transport->close();
