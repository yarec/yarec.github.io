cmd_svn


# 复制修改且未提交的文件到　/tmp/tt　目录下：
svn st|sed -e '/^?.*$/d'|sed -e 's/^[MA]/cp --parents/g'|sed -e 's/$/ \/tmp\/tt/' | sh

# 列出指定版本这间修改过的所有文件
svn log -r 724:HEAD -v|egrep '\.php|\.js|\.css|\.jpg|\.png|\.gif'|sort|uniq|sed 's@^\s*M\s*@.@'|sed 's@^\s*A\s*@.@'|sed '/^\s*D.*$/d'

# check out trunk from github 
svn co --depth empty https://github.com/${username}/${project}
cd ${project}
svn up trunk
