(define (cart lst)
  ;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ;;;;  product-tree                    ;;;;;;
  ;;;; '(a b) => ((a (b)) (b (a)))      ;;;;;;
  ;;;; '(a b c) => ((a (b (c)) (c (b))) ;;;;;;
  ;;;;              (b (a (c)) (c (a))) ;;;;;;
  ;;;               (c (a (b)) (b (a))));;;;;;
  ;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  (define (product-tree lst)
    (if (symbol? lst)
    lst
    (map (lambda (ls)
           (let ((tail (remv ls lst)))
         (cons ls (product-tree tail))))
         lst)))

  ;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ;;; leaf?                   ¡¡¡¡¡¡¡¡          ;;;
  ;;; '(x) is a leaf,reference the product-tree ;;;
  ;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  (define (leaf? ls)
    (and (pair? ls) (eqv? 1 (length ls))))

  ;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ;;; product-result                                      ;;;;
  ;;; walk through the tree, append the node in "result"  ;;;;
  ;;; if it's a leaf then get one result, and append it in;;;;
  ;;; the "sum"                                           ;;;;
  ;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  (define (product-result ls result)
    (cond ((leaf? ls)  ;; is the leaf and get one result
       (set! sum  (append sum (list (append result (list (car ls)))))))
      (else
       (map (lambda (x) ;;; map the child tree
          (product-result x (append result (list (car ls)))))   ;;; walk next tree and append the node to "result"
        (cdr ls)))))
;  (let ((sum '()))
  (define sum '())
  (map (lambda (x)  ;; walk every tree in list product by
     (product-result x '()) )        ;; product-tree
       (product-tree lst))
  sum)

(define (print l)
  (for-each
   (lambda (x)
     (display x)(newline))
   l))

(print (cart '(a b c d e f g)))
