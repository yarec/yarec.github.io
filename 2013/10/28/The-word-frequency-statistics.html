统计单词频率


#!/usr/bin/env bash
#Name: word_freq.sh  
#Description: Find out frequency of words in a file  
#Usage: ./word_freq.sh words_file.txt|sort -n -k 2
  
if [ $# -ne 1 ];
then
    echo "Usage: $0 filename";
    exit -1
fi
  
filename=$1
  
egrep -o "\b[[:alpha:]]+\b" $filename | \
awk '{ count[$0]++ } END{ printf("%-28s%s\n","Word","Count") ; \
 for(ind in count) { printf("%-28s%d\n",ind,count[ind]); } }'
