vsftpd-install


 

       logins.txt

tom
foo

        /etc/pam.d/ftp

auth required /lib/security/pam_userdb.so db=/etc/vsftpd/login
account required /lib/security/pam_userdb.so db=/etc/vsftpd/login

       /etc/vsftpd/vsftpd.conf

anonymous_enable=NO
local_enable=YES
write_enable=YES
anon_upload_enable=YES
anon_mkdir_write_enable=YES
anon_other_write_enable=YES
chroot_local_user=YES
guest_enable=YES
guest_username=virtual
listen=YES
listen_port=21
pasv_min_port=30000
pasv_max_port=30999
anon_world_readable_only=NO #for ls


执行下列命令：

db_load -T -t hash -f logins.txt /etc/vsftpd/login.db
chmod 600 /etc/vsftpd/login.db
/usr/sbin/useradd -d /home/ftpsite virtual
ls -ld /home/ftpsite

 

selinux 问题
"chcon -R -t public_content_t /var/ftp" 
man ftpd_selinux

 

setsebool -P ftpd_disable_trans 1
service vsftpd restart


iptables -A INPUT -i eth1 -p tcp --dport 30000:30999 -j ACCEPT #ftp

yum install db4 db4-devel 或者试试 /usr/lib/rpm/rpmdb_load
