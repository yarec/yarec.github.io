非root安装node


通常的linux环境是不会直接提供root用户，这种情况下想要安装node其实也很简单。只要在configure里指定PREFIX就可以了。
下面是完整的安装命令：

git clone https://github.com/joyent/node.git -b master
cd node
mkdir ~/opt
export PREFIX=~/opt; ./configure
make
make install
echo 'export PATH=~/opt/bin:${PATH}' >> ~/.bashrc
source ~/.bashrc

 测试一下：

$ node -v
v0.11.5-pre
$ npm -v
1.3.4

再测试一下webapp，以下代码保存到app.js

var http = require('http');
server = http.createServer(function (req, res) {
res.writeHeader(200, {"Content-Type": "text/plain"});
res.end("Hello Node");
})
server.listen(3000);

执行：node app.js
在浏览器中访问 http://127.0.0.1:3000
显示“Hello Node”说明一切正常。

试用Express:

npm install -g express
mkdir site && cd site
express .
cd . && npm install
node app.js

在浏览器中访问 http://127.0.0.1:3000


-> Welcome to Express

SICP的八皇后实现


八皇后问题：

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;一些实现基本功能的函数
;函数accumulate：在list各元素上累积运算------------------
(define (accumulate op initial seq)
  (if (null? seq)
    initial
    (op (car seq) (accumulate op initial (cdr seq)))))

;函数filter：list元素过滤
(define (filter predicate sequence)
  (cond ((null? sequence) '())
        ((predicate (car sequence))
         (cons (car sequence) (filter predicate (cdr sequence))))
        (else (filter predicate (cdr sequence)))))

;函数flatmap
(define (flatmap proc seq)
  (accumulate append '() (map proc seq)))

;函数enumerate-interval：产生list
(define (enumerate-interval low high)
  (if (> low high)
    '()
    (cons low (enumerate-interval (+ low 1) high))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;程序主要框架
(define (queens board-size)
  (define (queen-rows k)
    (if (= k 0)
      (list empty-board)
      (filter
        (lambda (positions) (safe? k positions))
        ;根据K值，过滤掉不符合条件的表
        ;将新的【board-size】个(k.new-col)分【board-size】次加入
        ;(queen-rows (- k 1))中的每个表项
        (flatmap
          (lambda (rest-of-queens)
            (map (lambda (new-col)
                   (adjoin-position new-col k rest-of-queens))
                 (enumerate-interval 1 board-size)))
          (queen-rows (- k 1))))))
  (show-all-solutions (queen-rows board-size) board-size 1))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;定义 empty-board
(define empty-board '())
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;函数adjoin-position：将新位置插进rest-of-queens头部
(define (adjoin-position new-col k rest-of-queens)
  (cons (cons k new-col) rest-of-queens))
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;check函数比较给出的new和exist位置是否在一行/列/斜线 是则返回#t  
;--for safe?
(define (check new exist)
  (or
    (= (car new) (car exist))
    (= (cdr new) (cdr exist))
    (= (abs (- (car new) (car exist)))
       (abs (- (cdr new) (cdr exist))))))
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;safe?函数 根据k值判断当前的positions是否符合要求
(define (safe? k positions)
  (define (danger? new)
    (check new (car positions)))
  (if (= k 1)
    #t
    (null? (filter danger? (cdr positions)))))
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;函数show-row输出一行
(define (show-row col board-size)
  (define (show-next col board-size count)
    ;一行的开始输出左括号
    (cond 
      ((= count 0)
       (begin (display "(") (show-next col board-size (+ count 1))))
      ((= count board-size) ;输入到末尾 判断该输出Q or * 输出并加右括号
       (if (= count col)
         (display "Q)")
         (display "*)")))
      ((= count col) ;第col列输出Q
       (begin (display "Q ") 
              (show-next col board-size (+ count 1))))
      (else ;其余输出*
        (begin (display "* ") 
               (show-next col board-size (+ count 1))))))
  (begin (show-next col board-size 0)
         (newline)))
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;show-all-solutions 输出所有解法
(define (show-all-solutions ls board-size count)
  (define (show-one list);输出一个解法
    (if (not(null? list))
      (begin
        (show-row (cdr (car list)) board-size)
        (show-one (cdr list)))
      '()))
  (begin
    (if (null? ls)
      (begin
        (newline)
        (display "共")
        (display (- count 1))
        (display "种解法,输出完毕"))
      (begin
        (display "Solution No.")
        (display count)
        (display ":")
        (newline)
        (newline)
        (+ count 1)
        (show-one (car ls))
        (newline)
        (show-all-solutions (cdr ls) board-size (+ count 1))))))

PlanBox-计划盒子


      PlanBox是我的一个web项目，其功能类似于云笔记，使用的技术包括perl、mongodb、extjs、catalyst（perl的mvc框架）。为了解决extjs库（压缩后1M多）加载慢的问题，将其封装到了chrome扩展。数据库使用的mongodb，程序部署在heroku（目前的免费服务已经够用了）。
      在云笔记还没有满天飞的年代，我就开始实现这个程序，以记载从网上搜集的各种知识。最开始使用的是java的经典框架（SSH），当时直接用的appfuse(后来国内有了springside)。界面用了echo(一个服务端的ajax框架，其nb之处在于有一个像vs studio的设计工具)，对于我这种对html和css不通的程序员来说，echo真的是一个神器啊。后来之所以换成了perl+extjs的方案，是因为想要把java放到线上成本是很大的。因为曾经用perl实现了一个脚本工具wt（work tool：旨在简化日常使用的命令行工具），对perl的精炼和强大深有体会。界面方面echo虽然强的不行，但唯一的缺陷是对java的绑定，于是另一个强大的ui框架extjs成了我折腾的对象。基本结构是extjs负责界面的显示和操作的处理，与后台perl程序通过REST以json格式来通信。经过一段时间的苦战，功能已经全部重写完成。planbox成为我重要的工具，记学习笔记、写工作计划和总结、写博客、生成博客文章...
      planbox和wt是我写过的对我来说最有用的代码，几乎天天都会用到。最初原型（webechonote）提交到googlecode的日期为Nov 6, 2009，数来也有四年多了，现在对之前的一些辛苦做一个小总结，也和大家分享一下softidy.com后台程序的故事。相信作为写代码的人，总有一些程序就像自己的孩子一样，永远是那么漂亮，愿意用最宝贵的时间去呵护它、培养它，即使它看上去总是笨手笨脚的。

太极经典


       老话讲“打拳先明理,理通拳法精”，学太极拳若不读读经典，细细琢磨，就算套路练的拈手即来，至多也就起到一个活动活动四肢的作用，这是任何一门运动都能达到的效果。所以作为中国的传统文化和宝贝，有必要认真的深入理解、发扬。本人初学太极，以下摘自互联网，备于此与君共勉。

       中国人对于著作常喜好托古，假借古人之名为作者，传承给后世的人。太极拳界重要的拳论有两篇，一是张三丰的《太极拳经》；一是王宗岳的《太极拳论》，虽然实际的作者有待考究，但是仍被太极拳界奉为圭皋，影响学习太极拳的人甚深。 值得注意的地方是，传统中国武术界门户之见甚多，而太极拳的支派亦甚众，然而所有的太极拳门派均不约而同地尊崇王宗岳《太极拳论》为太极拳界至文，足见其地位重要。

      《太极拳经》——张三丰
       一举动周身俱要轻灵，尤须贯串。气宜鼓荡，神宜内敛，无使有缺陷处，无使有凹凸处，无使有断续处。其根在脚，发于腿，主宰于腰，行于手指，由脚而腿而腰，总须完整一气，向前退后，乃能得机得势。有不得机得势处，身便散乱，其病必于腰腿求之，上下前后左右皆然。凡此皆是意，不在外面，有上即有下，有前则有后，有左则有右。如意要向上，即寓下意，若将物掀起而加以挫之之力。斯其根自断，乃攘／坏之速而无疑。虚实宜分清楚，一处有一处虚实，处处总此一虚实，周身节节贯串，无令丝毫间断耳。
      长拳者，如长江大海，滔滔不绝也。掤、捋、挤、按、采、挒、肘、靠，此八卦也。进步、退步、左顾、右盼、中定，此五行也。掤、捋、挤、按，即乾、坤、坎、离、四正方也。采、挒、肘、靠，即巽、震、兑、艮、四斜角也。进退盼顾定，即金木水火土也，合之则为十三势也。
      原注云：此系武当山张三丰祖师遗论。欲天下豪杰延年益寿，不徒作技艺之末也。

      《太极拳论》——王宗岳
      太极者。无极而生。（动静之机。）阴阳之母也。动之则分。静之则合。无过不及。随曲就伸。人刚我柔谓之走。我顺人背谓之粘。动急则急应。动缓则缓随。虽变化万端。而理唯／为一贯。
      由著熟而渐悟懂劲。由懂劲而阶及神明。然非用／功力之久。不能豁然贯通焉。虚领顶劲。气沉丹田。不偏不倚。忽隐忽现。左重则左虚。右重则右杳。仰之则弥高。俯之则弥深。进之则愈长。退之则愈促。一羽不能加。蝇虫不能落。人不知我。我独知人。英雄所向无敌。盖皆由此而及也。
      斯技旁门甚多。虽势有区别。概不外乎壮欺弱。慢让快耳。有力打无力。手慢让手快。皆是先天自然之能。非关学力而有为也。察四两拨千斤之句。显非力胜。观耄耋能御众之形。快何能为。立如平／秤准。活似车轮。偏沉则随。双重则滞。每见数年纯功。不能运化者。率皆自为人制。双重之病未悟耳。欲避此病。须知阴阳。粘即是走。走即是粘。阴不离阳。阳不离阴。阴阳相济方为懂劲。懂劲后。愈练愈精。默识揣摩。渐至从心所欲。本是舍己从人。多误舍近求远。所谓差之毫厘。谬之千里。学者不可不详辨焉。是为论。

      夫人神好清，而心扰之；人心好静，而欲牵之。常能遣其欲，而心自静，澄其心而神自清。自然六欲不生，三毒消灭。所以不能者，为心未澄，欲未遣也。能遣之者，内观其心，心无其心；外观其形，形无其形；远观其物，物无其物。三者既悟，唯见於空；观空亦空，空无所空；所空既无，无无亦无；无无既无，湛然常寂；寂无所寂，欲岂能生？欲既不生，即是真静。真常应物，真常得性；常应常静，常清静矣。如此清静，渐入真道；既入真道，名为得道，虽名得道，实无所得；为化众生，名为得道；能悟之者，可传圣道。

抽象与程序设计


需要复合过程和复合数据的原因是为了提高在设计程序时所位于的概念层次，提高设计的模块性，增强语言的表达能力。
数据是什么？
      有理数的定义：由构造函数make-rat和选择函数numer、denom提供了有理数的构造和对分子、分母的获取，并且满足下面条件：
对于任意整数n和任意非零整数d，如果x是(make-rat n d)，那么：(numer x)/(denom x) = n/d 
      scm实现如下:

(define make-rat cons)
(define numer car)
(define denom cdr)

      通过这段代码可以看出，有理数的实现直接转接到scheme语言的三个基本过程cons、car、cdr。这三个过程是scheme构建复合数据的基础。由于有理数的定义和cons等过程的含义相吻合，所以直接改个名子就ok了。
      序对的过程性表示：

(define (cons x y)
  (define (dispatch m)
    (cond ((= m 0) x)
          ((= m 1) y)
          (else (error "Argument not 0 or 1 -- CONS" m))))
  dispatch)

(define (car z) (z 0))

(define (cdr z) (z 1))

cons反回的并不是一个数据结构，而是一个过程。这种方式提供了一种表示复合数据的能力，在程序设计中扮演着一种核心角色，有关的程序设计风格通常称为消息传递。
有理数的相关操作：

(define (add-rat x y)
  (make-rat (+ (* (numer x) (denom y))
               (* (numer y) (denom x)))
            (* (denom x) (denom y))))

(define (sub-rat x y)
  (make-rat (- (* (numer x) (denom y))
               (* (numer y) (denom x)))
            (* (denom x) (denom y))))

(define (mul-rat x y)
  (make-rat (* (numer x) (numer y))
            (* (denom x) (denom y))))

(define (div-rat x y)
  (make-rat (* (numer x) (denom y))
            (* (denom x) (numer y))))

(define (equal-rat? x y)
  (= (* (numer x) (denom y))
     (* (numer y) (denom x))))

对有理数提供加、减、乘、除、相等？操作给应用程序，这样应用程序就可以在一个高层有理数概念上去构建程序。

这就是数据抽象带来的好处，只要高层的add-rat、make-rat等不变，对于具体是怎么实现的可以不用关心。

平方根求值算法


题目：求方程 x2 = S 的正数解

牛顿法：求S的平方根，先假设X0 = 1，然后根据以下公式求出X1，再将X1代入公式右边，继续求出X2 ...，通过有效次迭代后即可求出S的平方根，Xn+1：



当S=2时，

猜想值x 中间值S/xn 下一个猜想值Xn+1
1 (2/1) = 2 ((2 + 1)/2) = 1.5
1.5 (2/1.5) = 1.3333 ((1.3333 + 1.5)/2) = 1.4167
1.4167 (2/1.4167) = 1.4118 ((1.4167 + 1.4118)/2) = 1.4142
1.4142 ... ...

程序实现：

Js 版本

function sqrt_iter(g, y) { 
    if (Math.abs(g*g - y) < 0.000000001) { 
        return g;
    } else { 
        return sqrt_iter( (g + y / g)/2, y );  
    } 
}
function sqrt(y) { 
    return sqrt_iter(1.0, y);
}
console.log(sqrt(9));

scm版本

(define (square x) (* x x))
(define (average x y)
   (/ (+ x y) 2))
(define (sqrt x)
  (define (good-enough? guess)
    (< (abs (- (square guess) x)) 0.001))
  (define (improve guess)
    (average guess (/ x guess)))
  (define (sqrt-iter guess)
    (if (good-enough? guess)
        guess
        (sqrt-iter (improve guess))))
  (sqrt-iter 1.0))
(sqrt 9)

fixed-point版本

;公共部分
(define (square x) (* x x))
(define tolerance 0.00001)
(define (fixed-point f first-guess)
  (define (close-enough? v1 v2)
    (< (abs (- v1 v2)) tolerance))
  (define (try guess)
    (let ((next (f guess)))
      (if (close-enough? guess next)
          next
          (try next))))
  (try first-guess))
;(fixed-point cos 1.0)<br>;搜寻g在transform下的不动点
(define (fixed-point-of-transform g transform guess)
  (fixed-point (transform g) guess))

;----------------------------
(define (average x y)
   (/ (+ x y) 2))
(define (average-damp f)
  (lambda (x) (average x (f x))))

;平方根既是：搜寻 y->x/y 在平均阻尼下的不动点
(define (sqrt x) 
  (fixed-point-of-transform (lambda (y) (/ x y))
                            average-damp
                            1.0))

;立方根既是：搜寻 y->x/(y*y) 在平均阻尼下的不动点
(define (cube-root x) 
  (fixed-point-of-transform (lambda (y) (/ x (square y)))
                            average-damp
                            1.0))
;(sqrt 9)
(cube-root 27)

Newton's method

;公共部分
(define (square x) (* x x))
(define tolerance 0.00001)
(define (fixed-point f first-guess)
  (define (close-enough? v1 v2)
    (< (abs (- v1 v2)) tolerance))
  (define (try guess)
    (let ((next (f guess)))
      (if (close-enough? guess next)
          next
          (try next))))
  (try first-guess))
;(fixed-point cos 1.0)<br>;搜寻g在transform下的不动点
(define (fixed-point-of-transform g transform guess)
  (fixed-point (transform g) guess))

;----------------------------
(define dx 0.00001)
(define (deriv g)
  (lambda (x)
    (/ (- (g (+ x dx)) (g x))
       dx)))
(define (newton-transform g)
  (lambda (x)
    (- x (/ (g x) ((deriv g) x)))))
 
;平方根既是：搜寻 y->y*y-x 在牛顿变换下的不动点
(define (sqrt x)
  (fixed-point-of-transform (lambda (y) (- (square y) x))
                            newton-transform
                            1.0))
(sqrt 9)


过程抽象：

上面Js版本描述了一个模型，即牛顿法的递归函数。
scm版本与Js版类似，只不过这个版本抽象出了几个更基本的函数。
多了这一层的抽象，对于解平方根只起到了一个语义清晰的作用，某些时候这种做法会导致性能的降低。
然而对一个复杂系统来说，其内部可以有成千上万的模型，对于这种复杂度的控制，抽象是最有效的工具。
通过抽象，将基本的模型组织成高级的模型，再用高级模型再组织成更高级的模型，这样在高层次思考问题时复杂度就降低了。
优秀的程序设计者应该能够对系统的复杂度充分了解，并设计出具有适当抽象层次的程序模型结构。
fixed-point版本使用了更多的抽象层次，sqrt > fixed-point-of-transform > fixed-point > average-damp > average 。
Newton's method以另外一个模型来构建这一功能，sqrt > fixed-point-of-transform > fixed-point > newton-transform > deriv
